public static int dayOfWeek(int month, int day, int year)
{
  int weekday = firstDayOfYear(year);
  int additionalDays = dayOfYear(month, day, year) - 1;

  for(int d = 1; d <= additionalDays; d++)
  {
    weekday++; //add one to weekday

    if(weekday == 7)
      weekday = 0; //set to 0
  }
        
  return weekday; 
}
